function main(splash, args)
	splash.private_mode_enabled = false

  url = args.url
  assert(splash:go(url))
  assert(splash:wait(3))

  read_mores = assert(splash:select_all('._35pfga6W._2HBN-k68._3LkX-HIr'))
  
 	for _, btn in ipairs(read_mores) do
    assert(btn:mouse_click())
  end
  
  splash:set_viewport_full()
  return {
    png = splash.png(),
    html = splash.html()
  }
end
